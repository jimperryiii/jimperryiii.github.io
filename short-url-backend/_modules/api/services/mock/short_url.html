[docs]    def create_short_url(self, short_url, username):
        """Returns a new short url created by authenticated user.

        :param short_url: Short url title and destination to be created
        :type short_url: str
        :param username: User id for authenticated user
        :type username: str
        :return: Newly created short url, HTTP response code
        :rtype: str, int
        """
        if not self._validate_short_url(short_url):
            return json.dumps('Invalid data'), 422

        payload = short_url
        payload['username'] = username
        short_url = self._mock_invoke('create-short-url', payload)
        if short_url:
            return json.dumps(short_url), 200
        else:
            return json.dumps('Short url not created'), 409

    def _validate_short_url(self, short_url):
        """Returns True if short url is valid."""
        if short_url and short_url['destination'] and short_url['title']:
            return True
        else:
            return False

    def _mock_invoke(self, function_name, payload):
        """Returns results after invoking mock AWS lambda function.

        :param function_name: Name of the mock lambda function to invoke
        :type function_name: str
        :param payload: Input payload expected by mock lambda function
        :type payload: dict
        :raise: Exception raised if function_name is not valid
        :return: Results from invoking mock lambda function
        :rtype: dict
        """
        functions = {
            'get-all-short-urls': self._mock_get_all,
            'create-short-url': self._mock_create,
            'get-short-url': self._mock_get,
            'update-short-url': self._mock_update,
            'delete-short-url': self._mock_delete
        }

        mock_function = functions.get(function_name,
                                      self._invalid_function_name)
        return mock_function(payload)

    def _mock_get_all(self, payload):
        """Returns all short urls created by authenticated user.

        :param payload: User id for authenticated user
        :type payload: dict
        :return: List of short urls created by authenticated user
        :rtype: list
        """
        short_urlsData = json.loads(self.short_urlsJSON)
        short_urlsFiltered = [x for x in short_urlsData if
                              x['username'] == payload['username']]
        return short_urlsFiltered

    def _mock_create(self, payload):
        """Returns a new short url created by authenticated user.

        :param payload: Short url to be created
        :type payload: dict
        :return: Newly created short url
        :rtype: dict
        """
        short_urlsData = json.loads(self.short_urlsJSON)
        short_url = payload
        short_url['id'] = self._generate_short_path()
        short_urlsData.append(short_url)
        self.short_urlsJSON = json.dumps(short_urlsData)
        return short_url

    def _mock_get(self, payload):
        """Returns a single short url created by authenticated user.

        :param payload: Id of short url
        :type payload: dict
        :return: Requested short url
        :rtype: dict
        """
        short_urlsData = json.loads(self.short_urlsJSON)
        for short_url in short_urlsData:
            if short_url['id'] == payload['id'] and \
                    short_url['username'] == payload['username']:
                return short_url

    def _mock_update(self, payload):
        """Returns a single short url updated by authenticated user.

        :param payload: Short url updates
        :type id: dict
        :return: Updated short url
        :rtype: dict
        """
        short_urlsData = json.loads(self.short_urlsJSON)
        for short_url in short_urlsData:
            if short_url['id'] == payload['id'] and \
                    short_url['username'] == payload['username']:
                short_url.update(payload)
                self.short_urlsJSON = json.dumps(short_urlsData)
                return short_url

    def _mock_delete(self, payload):
        """Returns 'Item deleted' when short url is deleted.

        :param payload: Id of short url to delete
        :type payload: str
        :return: 'Item deleted'
        :rtype: str
        """
        short_urlsData = json.loads(self.short_urlsJSON)
        for short_url in short_urlsData:
            if short_url['id'] == payload['id'] and \
                    short_url['username'] == payload['username']:
                short_urlsData.remove(short_url)
                self.short_urlsJSON = json.dumps(short_urlsData)
                return 'Item deleted'

    def _invalid_function_name(self, payload):
        """Raises exception for invalid function."""
        raise Exception('Invalid function name')

    def _generate_short_path(self, size=7,
                             chars=string.ascii_letters + string.digits):
        """Returns a random generated short url id.

        :param size: Length of short path string
        :type size: int, optional
        :param chars: Valid characters allowed in short path
        :type chars: str, optional
        :return: Random generated short path string
        :rtype: str
        """
        return ''.join(random.choice(chars) for _ in range(size))
