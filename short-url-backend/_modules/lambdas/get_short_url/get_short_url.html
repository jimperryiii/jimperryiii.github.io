import boto3
import re
import logging

logger = logging.getLogger()
logger.setLevel(logging.INFO)

event_validation = {
    'id': r"^([a-zA-Z0-9]{7})$",
    'username': r"^[\w_\-.]{5,12}$"
}

[docs]def get_short_url(id, username):
    """Returns a single short url created by authenticated user.

    :param id: Id of short url
    :type id: str
    :param username: User id for authenticated user
    :type username: str
    :return: Requested short url
    :rtype: str
    """
    dynamodb = boto3.resource('dynamodb')
    table = dynamodb.Table('short-urls')

    if not validate_input(id, username):
        return

    response = table.get_item(
        Key={
            'id': id,
            'username': username
        },
    )

    if 'Item' in response:
        return response['Item']




[docs]def validate_event(event):
    """Validates all keys in lambda event."""
    for key in event_validation.keys():
        if key not in event:
            return False
    return True




[docs]def validate_input(id, username):
    """Runs regex validation for all inputs.

    :param id: Short url id to be returned
    :type id: str
    :param username: User id for authenticated user
    :type username: str
    :return: True if all inputs are valid
    :rtype: bool
    """
    if re.fullmatch(event_validation['id'], id) is None or \
            re.fullmatch(event_validation['username'], username) is None:
        return False
    return True
