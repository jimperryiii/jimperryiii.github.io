[docs]    def get_redirect(self, location):
        """Returns a browser redirect to the provided location.

        :param location: Redirect URL
        :type location: str
        :return: HTTP response for browser redirect
        :rtype: dict
        """
        response = {
            "statusCode": 302,
            "statusDescription": "Found",
            "isBase64Encoded": False,
            "headers": {
            }
        }
        response['headers']['Location'] = location
        return response
